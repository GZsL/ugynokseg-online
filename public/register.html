document.getElementById("registerForm").addEventListener("submit", async (e) => { e.preventDefault(); const name = document.getElementById("name").value.trim(); const email = document.getElementById("email").value.trim(); const password = document.getElementById("password").value.trim(); const res = await fetch("/api/auth/register", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ name, email, password }) }); const data = await res.json(); if (!res.ok) { alert(data.error || "Hiba történt"); return; } alert("Sikeres regisztráció!"); window.location.href = "/login.html"; });
